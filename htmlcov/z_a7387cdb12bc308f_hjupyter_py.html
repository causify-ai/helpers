22def run_notebook( 

23 file_name: str, 

24 scratch_dir: str, 

25 *, 

26 pre_cmd: str = "", 

27) -> None: 

28 """ 

29 Run jupyter notebook. 

30 

31 Assert if the notebook doesn't complete successfully. 

32 

33 :param file_name: path to the notebook to run. If this is a .py 

34 file, convert to .ipynb first 

35 :param scratch_dir: temporary dir storing the output 

36 :param pre_cmd: 

37 """ 

38 file_name = os.path.abspath(file_name) 

39 hdbg.dassert_path_exists(file_name) 

40 hio.create_dir(scratch_dir, incremental=True) 

41 # Build command line. 

42 cmd = [] 

43 if pre_cmd: 

44 cmd.append(f"{pre_cmd} &&") 

45 # Convert .py file into .ipynb if needed. 

46 root, ext = os.path.splitext(file_name) 

47 if ext == ".ipynb": 

48 notebook_name = file_name 

49 elif ext == ".py": 

50 cmd.append(f"jupytext --update --to notebook {file_name};") 

51 notebook_name = f"{root}.ipynb" 

52 else: 

53 raise ValueError(f"Unsupported file format for `file_name`='{file_name}'") 

54 # Execute notebook. 

55 cmd.append(f"cd {scratch_dir} &&") 

56 cmd.append(f"jupyter nbconvert {notebook_name}") 

57 cmd.append("--execute") 

58 cmd.append("--to html") 

59 cmd.append("--ExecutePreprocessor.kernel_name=python") 

60 # No time-out. 

61 cmd.append("--ExecutePreprocessor.timeout=-1") 

62 # Execute. 

63 cmd_as_str = " ".join(cmd) 

64 hsystem.system(cmd_as_str, abort_on_error=True, suppress_output=False)

106class NotebookImageExtractor: 

107 """ 

108 Extract marked regions from a Jupyter notebook, convert them to HTML and 

109 captures screenshots. 

110 

111 Initialize with input notebook path and output directory. 

112 """ 

113 

114 def __init__(self, notebook_path: str, output_dir: str) -> None: 

115 self.notebook_path = notebook_path 

116 self.output_dir = output_dir

118 def extract_regions_from_notebook(self) -> List[Tuple[str, str, List]]: 

119 """ 

120 Extract regions from a notebook based on extraction markers. 

121 

122 This function reads a Jupyter notebook and searches for all regions 

123 indicated by the markers inside cells: 

124 ``` 

125 # start_extract(mode)=<output_filename> 

126 ... 

127 # end_extract 

128 ``` 

129 Example: 

130 

131 1. To extract only the input code: 

132 # start_extract(only_input) = input_code.py 

133 ```python 

134 def test_func(): 

135 return "Test" 

136 ``` 

137 # end_extract 

138 

139 2. To extract only the output of code: 

140 # start_extract(only_output)=output.png 

141 ```python 

142 print("This is the output") 

143 ``` 

144 # end_extract 

145 

146 3. To extract both code and output: 

147 # start_extract(all) =full_output.html 

148 ```python 

149 print("This is both code and output") 

150 ``` 

151 # end_extract 

152 

153 :return: tuples (mode, out_filename, region_cells) for each extraction region. 

154 """ 

155 # Read notebook. 

156 nb = nbformat.read(self.notebook_path, as_version=4) 

157 start_marker_regex = re.compile( 

158 r""" 

159 \# # Match the literal '#' character (comment indicator) 

160 \s* # Allow optional whitespace after the '#' 

161 start_extract # Match the literal text 'start_extract' 

162 \( # Match the literal '(' (escaped since it's a special character) 

163 \s* # Allow optional whitespace after '(' 

164 (only_input|only_output|all) # Capture the extraction mode (one of: only_input, only_output, or all) 

165 \s* # Allow optional whitespace after the extraction mode 

166 \) # Match the literal ')' (escaped) 

167 \s* # Allow optional whitespace after ')' 

168 = # Match the literal '=' sign 

169 \s* # Allow optional whitespace after '=' 

170 (\S+) # Capture the output filename (one or more non-whitespace characters) 

171 """, 

172 re.VERBOSE, 

173 ) 

174 end_marker_regex = re.compile(r"#\s*end_extract\s*") 

175 regions = [] 

176 in_extract = False 

177 current_mode = None 

178 current_out_filename = None 

179 current_cells = [] 

180 for cell in nb.cells: 

181 if cell.cell_type != "code": 

182 continue 

183 if not in_extract: 

184 m = start_marker_regex.search(cell.source) 

185 if m: 

186 # A start marker was found. Capture the mode and output filename 

187 current_mode = m.group(1) 

188 current_out_filename = m.group(2) 

189 in_extract = True 

190 # Remove the start marker from the cell. 

191 cell.source = start_marker_regex.sub("", cell.source).strip() 

192 ## TODO: How can this happen? For me this is an error. 

193 # If there's also an end marker in this cell, complete the extraction region. 

194 if end_marker_regex.search(cell.source): 

195 cell.source = end_marker_regex.sub( 

196 "", cell.source 

197 ).strip() 

198 current_cells.append(cell) 

199 regions.append( 

200 (current_mode, current_out_filename, current_cells) 

201 ) 

202 current_cells = [] 

203 in_extract = False 

204 else: 

205 current_cells.append(cell) 

206 else: 

207 # If there's an end marker without a corresponding start, ignore it. 

208 pass 

209 else: 

210 # We are inside an extraction region, so continue adding cells to the region. 

211 if end_marker_regex.search(cell.source): 

212 cell.source = end_marker_regex.sub("", cell.source).strip() 

213 current_cells.append(cell) 

214 regions.append( 

215 (current_mode, current_out_filename, current_cells) 

216 ) 

217 current_cells = [] 

218 in_extract = False 

219 else: 

220 # If there's no end marker, just keep adding cells to the current region. 

221 current_cells.append(cell) 

222 if not regions: 

223 raise ValueError("No extraction markers found in the notebook.") 

224 return regions

227 self, nb: nbformat.NotebookNode, output_html: str

229 """

240 def capture_screenshot( 

241 self, html_file: str, screenshot_path: str, *, timeout: int = 2000 

242 ) -> None: 

243 """ 

244 Capture a screenshot of an HTML file using Playwright. 

245 

246 This function launches a headless Chromium browser, opens the 

247 provided HTML file, waits for a specified timeout to ensure the 

248 page is fully rendered, and then takes a screenshot saving it to 

249 the provided screenshot path. 

250 

251 :param html_file: path to the HTML file. 

252 :param screenshot_path: path where the screenshot will be saved. 

253 :param timeout: time in milliseconds to wait for the page to 

254 render. 

255 """ 

256 # Import playwright only when this function is called. 

257 from playwright.sync_api import sync_playwright 

258 

259 file_url = "file:///" + os.path.abspath(html_file) 

260 with sync_playwright() as p: 

261 # Launch a headless Chromium browser. 

262 browser = p.chromium.launch(headless=True) 

263 # Open the HTML file. 

264 page = browser.new_page(viewport={"width": 1200, "height": 800}) 

265 page.goto(file_url) 

266 # Wait for a specified timeout to ensure the page. 

267 page.wait_for_timeout(timeout) 

268 # Take a screenshot, saving to file. 

269 page.screenshot(path=screenshot_path) 

270 browser.close()

272 def _extract_and_capture(self) -> list: 

273 """ 

274 Extract notebook regions, convert each to HTML, and capture separate 

275 screenshots. 

276 

277 The function orchestrates the extraction of all marked regions 

278 from a Jupyter notebook. It processes each region independently: 

279 adjusting cells according to its extraction mode, converting the 

280 region to an HTML file, capturing a screenshot using Playwright, 

281 and then cleaning up the temporary HTML file. Screenshots are 

282 saved in the "screenshots" folder with filenames based on the 

283 name provided in the extraction marker. If a name is repeated, a 

284 counter suffix (_1, _2, etc.) is appended to ensure unique 

285 filenames. A list of screenshot file paths is returned. 

286 

287 :return: list of paths to the screenshot files. 

288 """ 

289 regions = self.extract_regions_from_notebook() 

290 screenshot_files = [] 

291 # Create screenshots folder if it doesn't exist. 

292 screenshots_folder = self.output_dir 

293 hio.create_dir(screenshots_folder, incremental=True) 

294 # Keep track of filename usage to handle duplicates. 

295 filename_counter: Dict[str, int] = {} 

296 # Process each region. 

297 for mode, out_filename, cells in regions: 

298 # Adjust each cell in the region according to the extraction mode. 

299 for cell in cells: 

300 if mode == "only_input": 

301 cell.outputs = [] 

302 elif mode == "only_output": 

303 cell.source = "" 

304 # Create a new notebook for the region. 

305 new_nb = nbformat.v4.new_notebook(cells=cells) 

306 temp_html = "temp_extract.html" 

307 self.convert_notebook_to_html(new_nb, temp_html) 

308 # Determine the final screenshot filename. 

309 base, ext = os.path.splitext(out_filename) 

310 if ext == "": 

311 ext = ".png" 

312 final_name = out_filename 

313 if final_name in filename_counter: 

314 filename_counter[final_name] += 1 

315 final_name = f"{base}_{filename_counter[out_filename]}{ext}" 

316 else: 

317 filename_counter[final_name] = 1 

318 screenshot_path = os.path.join(screenshots_folder, final_name) 

319 self.capture_screenshot(temp_html, screenshot_path) 

320 os.remove(temp_html) 

321 screenshot_files.append(screenshot_path) 

322 _LOG.info("Saved screenshot to %s", screenshot_path) 

323 return screenshot_files
