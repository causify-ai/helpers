306def _get_parquet_tiles_from_file_path(file_path: str) -> List[Tuple[str, Any]]: 

307 """ 

308 Hacky function to help get tile values from parquet file path. 

309 

310 Used by from_parquet when loading first n rows of a dataset only. 

311 

312 Example 

313 input: ...ccxt/binance/v1_0_0/currency_pair=CTK_USDT/ 

314 year=2023/month=3/26dc59f62b87403d9a3e9f04c7c21382-0.parquet 

315 output: [("currency_pair", "CTK_USDT"), ("year", 2023), ("month", 3)] 

316 """ 

317 path_parts = file_path.split("/") 

318 tiles = [] 

319 for part in path_parts: 

320 if "=" in part: 

321 col, value = part.split("=") 

322 value = int(value) if value.isdigit() else value 

323 tiles.append((col, value)) 

324 return tiles

541def _yield_parquet_tile( 

542 file_name: str, 

543 columns: List[str], 

544 filters: List[Any], 

545 asset_id_col: str, 

546) -> Iterator[pd.DataFrame]: 

547 """ 

548 Yield Parquet data in a single tile given the filters. 

549 

550 It is assumed that data is partitioned by asset_id, year and month, i.e. 

551 the file layout is: 

552 

553 ``` 

554 file_name/ 

555 asset_id=1032127330/ 

556 year=2021/ 

557 month=12/ 

558 data.parquet 

559 year=2022/ 

560 month=01/ 

561 data.parquet 

562 ... 

563 asset_id=2133227690/ 

564 year=2021/ 

565 month=12/ 

566 data.parquet 

567 year=2022/ 

568 month=01/ 

569 data.parquet 

570 ``` 

571 

572 :param file_name: see `from_parquet()` 

573 :param columns: see `from_parquet()` 

574 :param filters: see `from_parquet()` 

575 :param asset_id_col: name of the column with asset ids 

576 :return: a generator of `from_parquet()` dataframe 

577 """ 

578 # Without the schema being provided `pyarrow` incorrectly infers 

579 # type of the asset id column, i.e. `pyarrow` reads assets as 

580 # strings instead of integers. See the related discussion at 

581 # `https://issues.apache.org/jira/browse/ARROW-6114`. 

582 int_type = np.int64 

583 pyarrow_int_type = pa.from_numpy_dtype(int_type) 

584 schema = [ 

585 (asset_id_col, pyarrow_int_type), 

586 # TODO(Grisha): consider passing year and month column names as params. 

587 ("year", pyarrow_int_type), 

588 ("month", pyarrow_int_type), 

589 ] 

590 tile = from_parquet( 

591 file_name, 

592 columns=columns, 

593 filters=filters, 

594 schema=schema, 

595 ) 

596 hpandas.dassert_series_type_is(tile[asset_id_col], int_type) 

597 yield tile

685def build_year_month_filter( 

686 start_date: datetime.date, 

687 end_date: datetime.date, 

688) -> list: 

689 """ 

690 Use the year/months to build a Parquet filter. 

691 

692 If `start_date.year == end_date.year`, then return a list of 

693 three tuples (to be "ANDed" together) based on the year and months. 

694 Else, return a list of list of tuples: 

695 - the inner lists consist of AND filters; the inner lists are ORed 

696 together if used as a single filter 

697 - each inner list filter represents a calendar year or part thereof 

698 

699 One use case of this function is to generate a filter whose OR 

700 components can be processed one-by-one. For example, if memory constraints 

701 prevent loading an entire tile at once, then one could instead attempt to 

702 load one-year tiles one at a time. 

703 

704 NOTE: `start_date.day` and `end_date.day` are ignored. 

705 

706 TODO(Paul): Consider adding a switch to support smaller AND filter chunks 

707 (e.g., at monthly instead of yearly granularity). 

708 """ 

709 hdbg.dassert_isinstance(start_date, datetime.date) 

710 hdbg.dassert_isinstance(end_date, datetime.date) 

711 hdbg.dassert_lte(start_date, end_date) 

712 start_year = start_date.year 

713 end_year = end_date.year 

714 filter_ = [] 

715 # 

716 if start_year == end_year: 716 ↛ 717line 716 didn't jump to line 717 because the condition on line 716 was never true

717 filter_.append(("year", "==", start_year)) 

718 filter_.append(("month", ">=", start_date.month)) 

719 filter_.append(("month", "<=", end_date.month)) 

720 else: 

721 start_year_filter = [] 

722 start_year_filter.append(("year", "==", start_year)) 

723 start_year_filter.append(("month", ">=", start_date.month)) 

724 end_year_filter = [] 

725 end_year_filter.append(("year", "==", end_year)) 

726 end_year_filter.append(("month", "<=", end_date.month)) 

727 filter_.append(start_year_filter) 

728 filter_.append(end_year_filter) 

729 for year in range(start_year + 1, end_year): 729 ↛ 730line 729 didn't jump to line 730 because the loop on line 729 never started

730 year_filter = [] 

731 year_filter.append(("year", "==", year)) 

732 filter_.append(year_filter) 

733 return filter_

736def build_filter_with_only_equalities( 

737 start_timestamp: pd.Timestamp, end_timestamp: pd.Timestamp 

738) -> list: 

739 """ 

740 Build a list of Parquet filters based on equality conditions for partition 

741 columns. 

742 

743 This function creates a filter for each partition column (year, month, day) based on the 

744 equality conditions between components of the timestamp arguments when possible. 

745 

746 Example: 

747 Input args: 

748 start_timestamp: 2022-08-31T00:01:00+00:00 

749 end-timestamp: 2022-08-31T23:59:59+00:00

751 [("year", "=", 2022), ("month", "=", 8), ("day", "=", 31)]

766 if start_timestamp.day == end_timestamp.day:

771def collate_parquet_tile_metadata( 

772 path: str, 

773) -> pd.DataFrame: 

774 """ 

775 Report stats in a dataframe on Parquet file partitions. 

776 

777 The directories should be of the form `lhs=rhs` where "rhs" is a string 

778 representation of an `int`. 

779 

780 :param path: path to top-level Parquet directory 

781 :return: dataframe with two file size columns and a multiindex reflecting 

782 the Parquet path structure. 

783 """ 

784 hdbg.dassert_dir_exists(path) 

785 # Remove the trailing slash to simplify downstream accounting. 

786 if path.endswith("/"): 

787 path = path[:-1] 

788 hdbg.dassert(not path.endswith("/")) 

789 # Walk the path. 

790 # os.walk() yields a 3-tuple of the form 

791 # (dirpath: str, dirnames: List[str], filenames: List[str]) 

792 start_depth = len(path.split("/")) 

793 headers_set = set() 

794 dict_ = collections.OrderedDict() 

795 for triple in os.walk(path): 

796 # If the walk has taken us to, e.g., 

797 # asset_id=100/year=2010/month=1/data.parquet 

798 # then we expect 

799 # lhs = ("asset_id", "year", "month") 

800 # rhs = (100, 2010, 1) 

801 lhs, rhs = _process_walk_triple(triple, start_depth) 

802 # If the walkabout has not yet taken us to a file, continue. 

803 if not lhs: 

804 continue 

805 # The tuple `lhs` is to become the index headers. We check later 

806 # for uniqueness. 

807 headers_set.add(lhs) 

808 # Get the file name and full path. 

809 file_name = triple[2][0] 

810 file_path = os.path.join(triple[0], file_name) 

811 # Record the size of the file. We keep this in bytes for easy 

812 # join aggregations. 

813 size_in_bytes = os.path.getsize(file_path) 

814 dict_[rhs] = size_in_bytes 

815 # Ensure that headers are unambiguous. 

816 hdbg.dassert_eq(len(headers_set), 1) 

817 # Convert to a multiindexed dataframe. 

818 df = pd.DataFrame(dict_.values(), index=dict_.keys()) 

819 df.rename(columns={0: "file_size_in_bytes"}, inplace=True) 

820 headers = headers_set.pop() 

821 df.index.names = headers 

822 df.sort_index(inplace=True) 

823 # Add a more human-readable file size column. Keep the original numerical 

824 # one for downstream aggregations. 

825 file_size = df["file_size_in_bytes"].apply(hintros.format_size) 

826 df["file_size"] = file_size

835) -> Tuple[Tuple[str, ...], Tuple[int, ...]]: 

836 """ 

837 Process a triple returned by `os.walk()`

840 :param start_depth: the "depth" of `path` used in the call

845 rhs_vals: List[int] = [] 

846 # If there are subdirectories, do not process. 

847 if triple[1]: 

848 return tuple(lhs_vals), tuple(rhs_vals)

856 rhs_vals = []

858 lhs, rhs = string.split("=") 

859 lhs_vals.append(lhs) 

860 rhs_vals.append(int(rhs)) 

861 hdbg.dassert_eq(len(lhs_vals), len(rhs_vals)) 

862 return tuple(lhs_vals), tuple(rhs_vals)

883def get_parquet_filters_from_timestamp_interval( 

884 partition_mode: str, 

885 start_timestamp: Optional[pd.Timestamp], 

886 end_timestamp: Optional[pd.Timestamp], 

887 *, 

888 additional_filters: Optional[List[ParquetFilter]] = None, 

889) -> Union[ParquetOrAndFilter, ParquetAndFilter]: 

890 """ 

891 Convert a constraint on a timestamp [start_timestamp, end_timestamp] into a 

892 Parquet filters expression, based on the passed partitioning / tiling 

893 criteria. 

894 

895 :param partition_mode: control filtering of Parquet datasets. It needs to be 

896 in sync with the way the data was saved 

897 :param start_timestamp: start of the interval. `None` means no bound 

898 :param end_timestamp: end of the interval. `None` means no bound 

899 :param additional_filters: AND conditions to add to the final filter. 

900 E.g., if we want to constraint also on `exchange_id` and 'currency_pair`, 

901 we can specify 

902 `[("exchange_id", "in", (...)),("currency_pair", "in", (...))]` 

903 :return: list of OR-AND predicates 

904 """ 

905 # Check timestamp interval. 

906 left_close = True 

907 right_close = True 

908 hdateti.dassert_is_valid_interval( 

909 start_timestamp, 

910 end_timestamp, 

911 left_close=left_close, 

912 right_close=right_close, 

913 ) 

914 or_and_filter = [] 

915 if partition_mode == "by_year_month": 

916 # Handle the first and last year of the interval. 

917 if start_timestamp: 

918 # `[('year', '==', 2020), ('month', '>=', 6)]` 

919 and_filter = [ 

920 ("year", "==", start_timestamp.year), 

921 ("month", ">=", start_timestamp.month), 

922 ] 

923 or_and_filter.append(and_filter) 

924 if end_timestamp: 

925 # `[('year', '==', 2021), ('month', '<=', 3)]` 

926 and_filter = [ 

927 ("year", "==", end_timestamp.year), 

928 ("month", "<=", end_timestamp.month), 

929 ] 

930 or_and_filter.append(and_filter) 

931 if start_timestamp and end_timestamp: 

932 number_of_years = len( 

933 range(start_timestamp.year, end_timestamp.year + 1) 

934 ) 

935 if number_of_years == 1: 

936 # For a one-year range, we overwrite the result with a single AND 

937 # statement, e.g., `[Jan 2020, Mar 2020]` corresponds to 

938 # `[[('year', '==', 2020), ('month', '>=', 1), ('month', '<=', 3)]]`. 

939 # Note that this interval is different from and OR-AND form as 

940 # `[[('year', '==', 2020), ('month', '>=', 1)], 

941 # [('year', '==', 2020), ('month', '<=', 3)]]` 

942 # since the first AND clause include months <= 3 and the second one 

943 # include months >= 1, and the OR corresponds to the entire year, 

944 # instead of the interval `[Jan 2020, Mar 2020]`. 

945 and_filter = [ 

946 ("year", "==", start_timestamp.year), 

947 ("month", ">=", start_timestamp.month), 

948 ("month", "<=", end_timestamp.month), 

949 ] 

950 or_and_filter = [and_filter] 

951 elif number_of_years > 2: 

952 # For ranges over two years, one OR statement is necessary to bridge 

953 # the gap between first and last AND statement. 

954 # `[('year', '>', 2020), ('year', '<', 2023)]` 

955 # Inserted in middle as bridge between AND statements. 

956 and_filter = [ 

957 ("year", ">", start_timestamp.year), 

958 ("year", "<", end_timestamp.year), 

959 ] 

960 or_and_filter.insert(1, and_filter) 

961 else: 

962 # For intervals of exactly two years the two AND conditions are 

963 # enough to select the desired period of time. 

964 pass 

965 elif len(or_and_filter) == 1: 

966 # Handle the case when exactly one of the interval bounds is passed, 

967 # e.g., [June 2020, None]. 

968 # In this case the first year was covered by the code above (i.e., 

969 # `year >= 2020 and month == 6`) and we need to specify the rest of 

970 # the years (i.e., `year > 2020`). 

971 operator = ">" if start_timestamp else "<" 

972 timestamp = start_timestamp if start_timestamp else end_timestamp 

973 extra_filter = [("year", operator, timestamp.year)] 

974 or_and_filter.append(extra_filter) 

975 else: 

976 # If there is no interval provided, leave empty `or_and_filter` as is. 

977 pass 

978 elif partition_mode == "by_year_week": 978 ↛ 981line 978 didn't jump to line 981 because the condition on line 978 was never true

979 # TODO(gp): Consider using the same approach above for months also here. 

980 # Partition by year and week. 

981 hdbg.dassert_is_not( 

982 end_timestamp, 

983 None, 

984 "Parquet backend can't determine the boundaries of the data", 

985 ) 

986 # Include last week in the interval. 

987 end_timestamp += pd.DateOffset(weeks=1) 

988 # Get all weeks in the interval. 

989 dates = pd.date_range( 

990 start_timestamp.date(), end_timestamp.date(), freq="W" 

991 ) 

992 for date in dates: 

993 year = date.year 

994 # https://docs.python.org/3/library/datetime.html#datetime.date.isocalendar 

995 weekofyear = date.isocalendar().week 

996 and_filter = [("year", "=", year), ("weekofyear", "=", weekofyear)] 

997 or_and_filter.append(and_filter) 

998 else: 

999 raise ValueError(f"Unknown partition mode `{partition_mode}`!") 

1000 if additional_filters: 

1001 hdbg.dassert_isinstance(additional_filters, list) 

1002 if or_and_filter: 

1003 # Append additional filters for every present timestamp filter. 

1004 or_and_filter = [ 

1005 additional_filters + and_filter for and_filter in or_and_filter 

1006 ] 

1007 else: 

1008 # If no timestamp filters are provided, use additional filters. 

1009 or_and_filter = additional_filters 

1010 _LOG.debug("or_and_filter=%s", str(or_and_filter)) 

1011 if len(or_and_filter) == 0: 

1012 # Empty list is not acceptable value for pyarrow dataset. 

1013 # Only logical expression or `None`. 

1014 or_and_filter = None 

1015 return or_and_filter
