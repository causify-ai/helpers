366def make_hashable(obj: Any) -> collections.abc.Hashable: 

367 """ 

368 Coerce `obj` to a hashable type if not already hashable. 

369 """ 

370 ret = None 

371 if isinstance(obj, collections.abc.Mapping): 

372 # Handle dict-like objects. 

373 new_object = copy.deepcopy(obj) 

374 for k, v in new_object.items(): 

375 new_object[k] = make_hashable(v) 

376 ret = tuple(new_object.items()) 

377 elif isinstance(obj, collections.abc.Iterable) and not isinstance(obj, str): 

378 # The problem is that `str` is both `Hashable` and `Iterable`, but here 

379 # we want to treat it like `Hashable`, i.e. return string as it is. 

380 # Same with `Tuple`, but for `Tuple` we want to apply the function 

381 # recursively, i.e. make every element `Hashable`. 

382 ret = tuple([make_hashable(element) for element in obj]) 

383 elif isinstance(obj, collections.abc.Hashable): 383 ↛ 387line 383 didn't jump to line 387 because the condition on line 383 was always true

384 # Return the object as is, since it's already hashable. 

385 ret = obj 

386 else: 

387 ret = tuple(obj) 

388 return ret

391def intersect_configs(configs: Iterable[crococon.Config]) -> crococon.Config: 

392 """ 

393 Return a config formed by taking the intersection of configs. 

394 

395 - Key insertion order is not taken into consideration for the purpose of 

396 calculating the config intersection 

397 - The key insertion order of the returned config will respect the key 

398 insertion order of the first config passed in 

399 """ 

400 # Flatten configs and convert to sets for intersection. 

401 # We create a list so that we can reference a flattened config later. 

402 flattened = [c.flatten() for c in configs] 

403 hdbg.dassert(flattened, "Empty iterable `configs` received.") 

404 # Obtain a reference config. 

405 # The purpose of this is to ensure that the config intersection respects a key 

406 # ordering. We also make this copy so as to maintain the original (not 

407 # necessarily hashable) values. 

408 reference_config = flattened[0].copy() 

409 # Make values hashable. 

410 for flat in flattened: 

411 for k, v in flat.items(): 

412 flat[k] = make_hashable(v) 

413 sets = [set(c.items()) for c in flattened] 

414 intersection_of_flattened = set.intersection(*sets) 

415 # Create intersection. 

416 # Rely on the fact that Config keys are of type `str`. 

417 intersection = crococon.Config() 

418 for k, v in reference_config.items(): 

419 if (k, make_hashable(v)) in intersection_of_flattened: 

420 intersection[k] = v 

421 return intersection

455def diff_configs(configs: Iterable[crococon.Config]) -> List[crococon.Config]: 

456 """ 

457 Diff `Config`s with respect to their common intersection. 

458 

459 :return: for each config `config` in `configs`, return a new `Config` consisting 

460 of the part of `config` not in the intersection of the configs 

461 """

463 configs = list(configs) 

464 # Find the intersection of all the configs. 

465 intersection = intersect_configs(configs)
