134def _to_single_line_cmd(cmd: Union[str, List[str]]) -> str: 

135 """ 

136 Convert a multiline command (as a string or list of strings) into a single 

137 line. 

138 

139 E.g., convert 

140 ``` 

141 IMAGE=.../amp:dev \ 

142 docker-compose \ 

143 --file devops/compose/docker-compose.yml \ 

144 --file devops/compose/docker-compose_as_submodule.yml \ 

145 --env-file devops/env/default.env 

146 ``` 

147 into 

148 ``` 

149 IMAGE=.../amp:dev docker-compose --file ... 

150 ``` 

151 """ 

152 if isinstance(cmd, list): 152 ↛ 153line 152 didn't jump to line 153 because the condition on line 152 was never true

153 cmd = " ".join(cmd) 

154 hdbg.dassert_isinstance(cmd, str) 

155 cmd = cmd.rstrip().lstrip() 

156 # Remove `\` at the end of the line. 

157 cmd = re.sub(r" \\\s*$", " ", cmd, flags=re.MULTILINE) 

158 # Use a single space between words in the command. 

159 # TODO(gp): This is a bit dangerous if there are multiple spaces in a string 

160 # that for some reason are meaningful. 

161 cmd = " ".join(cmd.split()) 

162 return cmd

165def to_multi_line_cmd(docker_cmd_: List[str]) -> str: 

166 r""" 

167 Convert a command encoded as a list of strings into a single command 

168 separated by `\`. 

169 

170 E.g., convert 

171 ``` 

172 ['IMAGE=*****.dkr.ecr.us-east-1.amazonaws.com/amp:dev', 

173 '\n docker-compose', 

174 '\n --file amp/devops/compose/docker-compose.yml', 

175 '\n --file amp/devops/compose/docker-compose_as_submodule.yml', 

176 '\n --env-file devops/env/default.env'] 

177 ``` 

178 into 

179 ``` 

180 IMAGE=*****.dkr.ecr.us-east-1.amazonaws.com/amp:dev \ 

181 docker-compose \ 

182 --file devops/compose/docker-compose.yml \ 

183 --file devops/compose/docker-compose_as_submodule.yml \ 

184 --env-file devops/env/default.env 

185 ``` 

186 """ 

187 # Expand all strings into single lines. 

188 _LOG.debug("docker_cmd=%s", docker_cmd_) 

189 docker_cmd_tmp = [] 

190 for dc in docker_cmd_: 

191 # Add a `\` at the end of each string. 

192 hdbg.dassert(not dc.endswith("\\"), "dc='%s'", dc) 

193 dc += " \\" 

194 docker_cmd_tmp.extend(dc.split("\n")) 

195 docker_cmd_ = docker_cmd_tmp 

196 # Remove empty lines. 

197 docker_cmd_ = [cmd for cmd in docker_cmd_ if cmd.rstrip().lstrip() != ""] 

198 # Package the command. 

199 docker_cmd_ = "\n".join(docker_cmd_) 

200 # Remove a `\` at the end, since it is not needed. 

201 docker_cmd_ = docker_cmd_.rstrip("\\") 

202 _LOG.debug("docker_cmd=%s", docker_cmd_) 

203 return docker_cmd_

353def _filter_existing_paths(paths_from_user: List[str]) -> List[str]: 

354 """ 

355 Filter out the paths to non-existent files. 

356 

357 :param paths_from_user: paths passed by user 

358 :return: existing paths 

359 """ 

360 paths = [] 

361 for user_path in paths_from_user: 

362 if user_path.endswith("/*"): 

363 # Get the files according to the "*" pattern. 

364 dir_files = glob.glob(user_path) 

365 if dir_files: 365 ↛ 367line 365 didn't jump to line 367 because the condition on line 365 was never true

366 # Check whether the pattern matches files. 

367 paths.extend(dir_files) 

368 else: 

369 _LOG.error(

371 "'%s' pattern doesn't match any files: " 

372 "the directory is empty or path does not exist" 

373 ), 

374 user_path, 

375 ) 

376 elif os.path.exists(user_path): 

377 paths.append(user_path) 

378 else: 

379 _LOG.error("'%s' does not exist", user_path) 

380 return paths
