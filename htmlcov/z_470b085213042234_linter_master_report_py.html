87def _calculate_exit_status( 

88 branch_dirty_status: bool, 

89 master_lints: int, 

90 branch_lints: int, 

91) -> Tuple[int, str]: 

92 """ 

93 Calculate status and error message. 

94 """ 

95 exit_status = 0 

96 errors = [] 

97 if branch_dirty_status: 

98 errors.append("**ERROR**: Run `linter.py. -b` locally before merging.") 

99 exit_status = 1 

100 if master_lints > 0: 

101 errors.append("**WARNING**: Your branch has lints. Please fix them.") 

102 if branch_lints > master_lints: 

103 exit_status = 1 

104 errors.append("**ERROR**: You introduced more lints. Please fix them.") 

105 return exit_status, "\n".join(errors)

123def _calculate_stats( 

124 base_commit_sha: str, 

125 head_commit_sha: str, 

126 head_branch_name: str, 

127 build_url: Optional[str] = None, 

128) -> Tuple[int, str]: 

129 """ 

130 Compute the statistics from the linter when run on a branch vs master. 

131 

132 :param base_commit_sha: hash of the branch to compare 

133 :param head_branch_name: name of the branch to be compared 

134 :param branch_name: branch name for the report 

135 :param build_url: jenkins build url 

136 :return: an integer representing the exit status and an error message. 

137 """ 

138 dir_name = "." 

139 # TODO(Sergey): Not sure what to do with files that are not present in the 

140 # branch. 

141 remove_files_non_present = False 

142 # Find the files that are modified in the branch. 

143 mod_files = hgit.get_modified_files_in_branch( 

144 dir_name, 

145 base_commit_sha, 

146 remove_files_non_present=remove_files_non_present, 

147 ) 

148 branch_lints, branch_dirty, linter_message = _perform_linter_for_test_branch( 

149 base_commit_sha 

150 ) 

151 master_lints, master_dirty = _perform_linter_for_reference_branch( 

152 base_commit_sha, mod_files 

153 ) 

154 ( 

155 master_dirty_status, 

156 branch_dirty_status, 

157 exit_status, 

158 errors, 

159 ) = _compute_stats(master_dirty, branch_dirty, master_lints, branch_lints) 

160 # Message 

161 message = list() 

162 # Report title 

163 message.append("# Results of the linter build") 

164 # Console url for Jenkins 

165 console_url = os.path.join(str(build_url), "consoleFull") 

166 if build_url is not None: 

167 console_message = f"Console output: {console_url}" 

168 else: 

169 console_message = "Console output: No console output" 

170 message.append(console_message) 

171 # Statuses and additional info 

172 message.append(f"- Master (sha: {base_commit_sha})") 

173 message.append(f"\t- Number of lints: {master_lints}") 

174 message.append(f"\t- Dirty (i.e., linter was not run): {master_dirty_status}") 

175 message.append(f"- Branch ({head_branch_name}: {head_commit_sha})") 

176 message.append(f"\t- Number of lints: {branch_lints}") 

177 message.append(f"\t- Dirty (i.e., linter was not run): {branch_dirty_status}") 

178 diff_lints = branch_lints - master_lints 

179 message.append( 

180 f"\nThe number of lints introduced with this change: {diff_lints}" 

181 ) 

182 # Format report in order. Message \n Errors \n Linter output. 

183 message = "\n".join(message) 

184 message += "\n\n" + errors 

185 message += "\n" + linter_message 

186 return exit_status, message
